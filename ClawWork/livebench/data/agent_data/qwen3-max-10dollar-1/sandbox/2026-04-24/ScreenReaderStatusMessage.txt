import React, { useEffect, useState } from 'react';
import './ScreenReaderStatusMessage.css';

interface ScreenReaderStatusMessageProps {
  message: string | React.ReactNode;
  visible?: boolean;
}

const ScreenReaderStatusMessage: React.FC<ScreenReaderStatusMessageProps> = ({ 
  message, 
  visible = false 
}) => {
  const [queue, setQueue] = useState<(string | React.ReactNode)[]>([]);

  useEffect(() => {
    if (message) {
      setQueue(prev => [...prev, message]);
      
      // Remove message from queue after a delay to allow screen reader to announce
      const timer = setTimeout(() => {
        setQueue(prev => {
          const newQueue = [...prev];
          newQueue.shift();
          return newQueue;
        });
      }, 1000);
      
      return () => clearTimeout(timer);
    }
  }, [message]);

  return (
    <div className="screen-reader-status-container" role="status" aria-live="polite">
      {visible && (
        <span aria-hidden="true">
          {message}
        </span>
      )}
      {queue.map((msg, index) => (
        <div key={index} className="screen-reader-status-message">
          {msg}
        </div>
      ))}
    </div>
  );
};

export default ScreenReaderStatusMessage;
